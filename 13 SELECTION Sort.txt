EXP:13 SELECTION Sort
#include <stdio.h>

int main() {
    int n, i, j, min;
    float a[50], temp;

    printf("Enter number of products: ");
    scanf("%d", &n);

    printf("Enter product prices:\n");
    for (i = 0; i < n; i++)
        scanf("%f", &a[i]);

    for (i = 0; i < n - 1; i++) {
        min = i;
        for (j = i + 1; j < n; j++)
            if (a[j] < a[min])
                min = j;

        temp = a[i];
        a[i] = a[min];
        a[min] = temp;
    }

    printf("\nSorted prices:\n");
    for (i = 0; i < n; i++)
        printf("%.2f ", a[i]);

    return 0;
}
